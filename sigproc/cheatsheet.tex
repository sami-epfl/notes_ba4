\section{Useful sequences}
\begin{defn}[title=Unit impulse | Delta]
$\delta[n] = \begin{cases}1 & n=0\\0 & n\ne 0\end{cases}$, \quad $\delta_m[n] = \delta[n-m]$
\end{defn}

\begin{defn}[title=Unit step]
$u[n] = \begin{cases}1 & n\ge 0\\0 & n<0\end{cases}$
\end{defn}

\begin{defn}[title=Exponential decay]
$x[n] = a^n u[n]$, \quad $0 < a < 1$
\end{defn}

\begin{defn}[title=Sinusoidal oscillation]
$x[n] = A \cos(\omega n + \phi)$, \quad $A\in\R$, $\omega\in\R$, $\phi\in\R$
\end{defn}

\begin{defn}[title=Complex exponential]
$x[n] = A\,e^{j(\omega n + \phi)}$, \quad $A\in\R$ amplitude, $\phi$ phase, $\omega\in\R$ frequency (rad).
\end{defn}

\begin{warn}[title=Key differences]
\begin{itemize}
  \item $\delta[n]$ is the most concentrated signal in time, $u[n]$ is the most concentrated in frequency.
  \item $\delta[n]$ has all frequencies equally present, $u[n]$ has only DC.
  \item $\delta[n]$ is the identity for convolution, $u[n]$ is the identity for multiplication.
\end{itemize}
\end{warn}

\begin{thm}[title=Energy and Power]
\begin{itemize}
  \item $E_{x} = \sum_{n=-\infty}^{\infty}|x[n]|^2$
  \item $P_{x} = \lim_{N\to\infty}\frac{1}{2N+1}\sum_{n=-N}^{N}|x[n]|^2$. 
  \item $E_{x}<\infty$ and $P_{x}=0$ $\Rightarrow$ Energy signal
  \item $P_{x}<\infty$ and $E_{x}=\infty$ $\Rightarrow$ Power signal
\end{itemize}
\end{thm}

\section{Basic operations}
\begin{defn}[title=Amplitude scaling]
$y[n] = \alpha x[n]$
\end{defn}
\begin{defn}[title=Sum of signals]
$y[n] = x_1[n] + x_2[n]$
\end{defn}
\begin{defn}[title=Product of signals]
$y[n] = x_1[n] \cdot x_2[n]$
\end{defn}
\begin{defn}[title=Time shift by k $\in \Z$]
$y[n] = \mathcal{S}^{k} x[n] = x[n+k]$
\end{defn}
\begin{defn}[title=Time reversal]
$y[n] = \mathcal{R} x[n] = \mathcal{S}^{-n} x[0] = x[-n]$
\end{defn}
\begin{thm}[title=Sum of a sequence]
$\sum_{k=m}^{n} z^{k} = \begin{cases} n-m+1 & z=1 \\ \frac{z^m-z^{n+1}}{1-z} & z\ne 1 \end{cases}$
\end{thm}
\section{Vector spaces of signals}

\begin{defn}[title=Vector space of signals]
\begin{itemize}
    \item $\ell_2(\mathbb{Z})$: space of square-summable, infinite sequences:
    \[
    x \in \ell_2(\mathbb{Z}) \;\Longleftrightarrow\; \|x\|^2 = \sum_{n \in \mathbb{Z}} |x[n]|^2 < \infty
    \]

    \item $L_2([a,b])$: space of square-integrable functions over the interval $[a,b]$:
    \[
    x \in L_2([a,b]) \;\Longleftrightarrow\; \int_a^b |x(t)|^2 \, dt < \infty
    \]
\end{itemize}
In both cases, we require vectors to have finite energy
\end{defn}

\begin{prop}[title=Inner product space]
  \begin{itemize}
    \item $\|x\| = \sqrt{\langle x, x \rangle}$
    \item $\langle x, y \rangle = \langle y, x \rangle^{*}$
    \item $\langle x, \alpha y \rangle = \alpha \langle x, y \rangle$
    \item $\langle \alpha x, y \rangle = \alpha^{*} \langle x, y \rangle$
  \end{itemize}
\end{prop}

\begin{prop}[title=Orthogonal basis coordinates]
For orthog.\ basis $B=\{\mathbf{v}_k\}_{k=0}^{N-1}$ with inner product $\langle\cdot,\cdot\rangle$, the coord.\ of $\mathbf{x}$ in $B$ are:
\[
  [\mathbf{x}]_B[k] = \frac{\langle \mathbf{v}_k,\, \mathbf{x}\rangle}{\langle \mathbf{v}_k,\, \mathbf{v}_k\rangle}, \quad\text{i.e. } \mathbf{x} = \sum_{k} [\mathbf{x}]_B[k]\,\mathbf{v}_k
\]
DFT: basis $\{\mathbf{w}_k\}$, $\|\mathbf{w}_k\|^2=N$, so $X[k]=\langle\mathbf{w}_k,\mathbf{x}\rangle$ (the $N$ factor is absorbed into the synthesis $1/N$).
\end{prop}

\section{The Complex Exponential}

\begin{defn}[title=Discrete-time complex exponential]
$w[n] = A\,e^{j(\omega n + \phi)}$, \quad $A\in\R$ amplitude, $\phi$ phase, $\omega\in\R$ frequency (rad).
\end{defn}

\begin{prop}[title=Recursive form]
$w[n+1] = w[n]\,e^{j\omega}$: each step rotates by $\omega$ on the unit circle.
\end{prop}

\subsection{Periodicity}
$\mathbf{w}$ is periodic iff $\exists N\in\Z^+, M\in\Z$ such that $\omega N = 2\pi M$:
\[
  \omega = \frac{M}{N}2\pi, \quad M,N\in\Z
\]
Only oscillations with \textbf{rational multiple of $2\pi$} are periodic.

\subsection{Aliasing}
\begin{warn}[title=Key identity]
$e^{j\theta} = e^{j(\theta + 2k\pi)}$ $\forall k\in\Z$. Hence:
\[
  e^{j(\omega+2k\pi)n} = e^{j\omega n} \quad \forall k\in\Z.
\]
Adding multiples of $2\pi$ to $\omega$ gives the \textbf{same samples}.
\end{warn}

\begin{itemize}
  \item $0\le\omega<\pi$: counterclockwise, increasing speed
  \item $\omega=\pi$: max speed, $+1/-1$ alternating
  \item $\pi<\omega<2\pi$: more ``economical'' as clockwise $\omega'=\omega-2\pi$
\end{itemize}
\textbf{Reference interval:} $[-\pi, \pi]$.

\subsection{Why complex oscillations?}
\begin{itemize}
  \item \kv{Simpler math}{$\cos(\alpha+\beta)=\Re\{e^{j\alpha}e^{j\beta}\}$}
  \item \kv{Compact notation}{polar vs.\ two real coordinates}
  \item \kv{Digital signals}{can natively be complex-valued}
\end{itemize}

\section{The DFT}

\subsection{\texorpdfstring{Fourier Basis for $\mathbb{C}^N$}{Fourier Basis for CN}}
\begin{defn}[title=Fourier basis vectors]
$w_k[n] = e^{j\frac{2\pi}{N}kn}$, \quad $n,k=0,\ldots,N-1$.\\
Harmonically related: $\omega_k = \frac{2\pi}{N}k$.\\
\textbf{Note:} $\mathbf{w}_k = \mathbf{w}_{N-k}^*$.
\end{defn}

\begin{thm}[title=Orthogonality]
$\langle \mathbf{w}_h, \mathbf{w}_k \rangle = N\,\delta[h-k]$, where
\[
  \langle \mathbf{x},\mathbf{y}\rangle = \sum_{n=0}^{N-1} x^*[n]\,y[n].
\]
Basis is orthogonal but \textbf{not} orthonormal (norms $= \sqrt{N}$).
\end{thm}

\subsection{Analysis \& Synthesis}
\begin{defn}[title=DFT (analysis)]
\[
  X[k] = \sum_{n=0}^{N-1} x[n]\,e^{-j\frac{2\pi}{N}nk}, \quad k=0,\ldots,N-1
\]
\end{defn}

\begin{defn}[title=Inverse DFT (synthesis)]
\[
  x[n] = \frac{1}{N}\sum_{k=0}^{N-1} X[k]\,e^{j\frac{2\pi}{N}nk}, \quad n=0,\ldots,N-1
\]
\end{defn}

\subsection{Elementary DFT Pairs}

\begin{prop}[title=Impulse \& constant (duality)]
$\boldsymbol{\delta} \overset{\mathrm{DFT}}{\longleftrightarrow} \mathbf{1}$\quad (most concentrated in time $\Rightarrow$ flat spectrum)\\
$\mathbf{1} \overset{\mathrm{DFT}}{\longleftrightarrow} N\boldsymbol{\delta}$\quad (most spread in time $\Rightarrow$ single freq.)
\end{prop}

\begin{prop}[title=Shifted impulse]
$\delta[n-m] \overset{\mathrm{DFT}}{\longleftrightarrow} e^{-j\frac{2\pi}{N}mk} = \mathbf{w}_m^*[k]$
\end{prop}

\begin{prop}[title=Harmonic sinusoid]
$\mathbf{w}_m \overset{\mathrm{DFT}}{\longleftrightarrow} N\boldsymbol{\delta}_m$, \quad where $w_m[n]=e^{j\frac{2\pi}{N}mn}$\\
Spread in time $\Rightarrow$ single nonzero freq.\ coefficient.
\end{prop}

\begin{prop}[title=Rectangular ($M$-length rect.)]
$x[n]=1$ for $0\le n < M$, else $0$:
\[
  X[k] = \frac{\sin(\pi Mk/N)}{\sin(\pi k/N)}\,e^{j\frac{\pi}{N}(M-1)k}
\]
Zeros at $k=mN/M$ ($m\ne 0$). $M=N$: gives $N\boldsymbol{\delta}$.
\end{prop}

\begin{warn}[title=Time--frequency concentration duality]
Most concentrated in time $\Leftrightarrow$ most spread in frequency (and vice versa). Generalizes $\boldsymbol{\delta}\leftrightarrow\mathbf{1}$ and $\mathbf{1}\leftrightarrow N\boldsymbol{\delta}$.
\end{warn}

\begin{prop}[title=Roots of unity]
The $N$ solutions of $z^N=1$: $z_k = e^{j\frac{2\pi}{N}k}$, $k=0,\ldots,N-1$. These are the frequencies $\omega_k=\frac{2\pi}{N}k$ satisfying $e^{j\omega_k N}=1$.
\end{prop}

\subsection{Matrix Form}
\begin{prop}[title=DFT matrix]
$\mathbf{X} = \mathbf{W}\mathbf{x}$, \quad $\mathbf{x} = \tfrac{1}{N}\mathbf{W}^H\mathbf{X}$, where $W_N = e^{-j\frac{2\pi}{N}}$:
\[
  \mathbf{W} =
  \begin{bmatrix}
    W_N^0 & W_N^0 & \cdots & W_N^0 \\
    W_N^0 & W_N^1 & \cdots & W_N^{N-1} \\
    \vdots & & \ddots & \vdots \\
    W_N^0 & W_N^{N-1} & \cdots & W_N^{(N-1)^2}
  \end{bmatrix}
\]
$\mathbf{W}^H\mathbf{W} = N\mathbf{I}$.
\end{prop}

\begin{warn}[title=Powers modulo $N$]
$W_N^n = W_N^{n\,\mathrm{mod}\,N}$ (aliasing property). First row and column are all 1 since $W_N^0=1$.
\end{warn}

\subsubsection{Small DFT matrices}
\[
  \mathbf{W}_2 = \begin{bmatrix}1&1\\1&-1\end{bmatrix}
\]
\[
  \mathbf{W}_3 = \begin{bmatrix}1&1&1\\1&W_3&W_3^2\\1&W_3^2&W_3\end{bmatrix},\quad W_3=\tfrac{-1-j\sqrt{3}}{2}
\]
\[
  \mathbf{W}_4 = \begin{bmatrix}1&1&1&1\\1&-j&-1&j\\1&-1&1&-1\\1&j&-1&-j\end{bmatrix}
\]

\section{ Properties}
\subsection{DFT Properties}

\begin{prop}[title=Linearity]
$\mathrm{DFT}\{a\mathbf{x}+b\mathbf{y}\} = a\,\mathrm{DFT}\{\mathbf{x}\}+b\,\mathrm{DFT}\{\mathbf{y}\}$, $a,b\in\mathbb{C}$
\end{prop}

\begin{prop}[title=Signal decomposition (computation trick)]
Decompose $\mathbf{x}=\sum_i a_i\mathbf{s}_i$ into signals with known DFTs, then use linearity.\\
\textbf{Ex.:} $\mathbf{x}=a\boldsymbol{\delta}+b\cdot\mathbf{1}$ $\Rightarrow$ $\mathbf{X}=a\cdot\mathbf{1}+bN\boldsymbol{\delta}$
\end{prop}

\begin{prop}[title=Circular time shift]
$x[(n-n_0)\!\mod N]\overset{\mathrm{DFT}}{\longleftrightarrow} e^{-j\frac{2\pi}{N}n_0 k}\,X[k]$
\end{prop}

\begin{prop}[title=Frequency shift]
$x[n]\,e^{j\frac{2\pi}{N}k_0 n}\overset{\mathrm{DFT}}{\longleftrightarrow} X[k-k_0]$
\end{prop}

\begin{prop}[title=Modulation by cosine]
$x[n]\cos\!\left(\tfrac{2\pi}{N}k_0 n\right)\overset{\mathrm{DFT}}{\longleftrightarrow}\tfrac{X[k-k_0]+X[k+k_0]}{2}$
\end{prop}

\begin{prop}[title=Time/frequency reversal]
$\mathrm{DFT}\{\mathcal{R}\mathbf{x}\}=\mathcal{R}\mathbf{X}$\\
$\mathrm{DFT}\{\mathbf{x}^*\}=\mathcal{R}\mathbf{X}^*$
\end{prop}

\begin{thm}[title=Parseval's theorem (DFT)]
\[
  \sum_{n=0}^{N-1}|x[n]|^2 = \frac{1}{N}\sum_{k=0}^{N-1}|X[k]|^2 = \|\mathbf{x}\|_2^2
\]
Energy is preserved across the DFT change of basis ($1/N$ from non-normalized basis). $|X[k]|^2/N$ = energy at frequency $\omega_k$. Useful: $\|\mathbf{x}-\mathbf{y}\|_2^2 = \frac{1}{N}\|\mathbf{X}-\mathbf{Y}\|_2^2$.
\end{thm}

\subsection{Symmetries of the DFT}
A signal $\mathbf{x}$ is \textbf{symmetric} if $x[n]=x[N-n]$.\\
A signal is \textbf{Hermitian-symmetric} if $x[n]=x^*[N-n]$.

\begin{thm}[title=Symmetry summary]
\renewcommand{\arraystretch}{1}
\begin{tabularx}{\linewidth}{@{}XX@{}}
\toprule
\textbf{Time domain} $x[n]$ & \textbf{Freq.\ domain} $X[k]$ \\
\midrule
Real & $X[-k]=X^*[k]$ \\
Real \& even & Real \& even \\
Real \& odd & Imaginary \& odd \\
Complex \& even & Complex \& even \\
Complex \& odd & Complex \& odd \\
\bottomrule
\end{tabularx}
\end{thm}

\section{Reading \& Plotting the DFT}

\begin{defn}[title=DFT magnitude plot]
Index $k$ encodes frequency $\omega_k = \frac{2\pi}{N}k$:
\begin{itemize}
  \item $k=0$: DC (zero frequency)
  \item $k=1\ldots\lfloor N/2\rfloor$: positive (CCW) frequencies, increasing
  \item $k=\lfloor N/2\rfloor$: maximum $\omega=\pi$ (fastest oscillation)
  \item $k=\lfloor N/2\rfloor+1\ldots N-1$: negative (CW) frequencies, decreasing
\end{itemize}
For real $\mathbf{x}$: often show only $k=0\ldots N/2$ (symmetric).
\end{defn}

\begin{warn}[title=DFT phase]
Phase is $2\pi$-periodic, plotted over $[-\pi,\pi]$. Values outside are \textbf{wrapped}. Phase offset $\theta$ in time shifts all components; same magnitude DFT can give different time-domain shapes.
\end{warn}

\begin{prop}[title=Frequency axis in Hz]
If signal sampled at $F_s = N/T$ Hz (T = duration in seconds):
\[
  k \to \frac{F_s}{N}k \text{ Hz}, \quad 0\le k\le N/2
\]
\[
  f_{\max} = \frac{F_s}{2} \text{ (Nyquist)}
\]
For $k>N/2$: $k\to \frac{F_s}{2}-\frac{F_s}{N}k$ Hz (negative freqs).
\end{prop}

\section{DFS \& DFT Periodicity}

\begin{thm}[title=DFT implies N-periodicity]
The inverse DFT formula is valid for \textbf{all} $n\in\Z$:
\[
  x[n] = \frac{1}{N}\sum_{k=0}^{N-1}X[k]\,e^{j\frac{2\pi}{N}nk} = x[n\!\mod N]
\]
Both $x$ and $X$ are implicitly $N$-periodic.
\end{thm}

\begin{defn}[title=DFS (Discrete Fourier Series)]
For an $N$-periodic signal $\tilde{x}$:
\[
  \tilde{\mathbf{X}} = \mathrm{DFS}\{\tilde{\mathbf{x}}\} = \mathrm{DFT}\{\tilde{x}[0],\ldots,\tilde{x}[N-1]\}
\]
DFS $\equiv$ DFT: same formulas, different interpretation (explicit periodicity).
\end{defn}

\begin{prop}[title=Circular shift (DFS)]
$\tilde{x}[n-n_0] \overset{\mathrm{DFS}}{\longleftrightarrow} e^{-j\frac{2\pi}{N}n_0 k}\,\tilde{X}[k]$
\end{prop}

\begin{warn}[title=Time shifts are circular]
With finite-length signals, time shifts wrap around modulo $N$. A linear phase multiplication on DFT coefficients computes the \textbf{circular shift} of the signal.
\end{warn}

\section{The DTFT}

\begin{defn}[title=DTFT (analysis)]
For infinite-length sequence $\mathbf{x}$:
\[
  X(\omega) = \sum_{n=-\infty}^{\infty} x[n]\,e^{-j\omega n}
\]
$X(\omega)$ is a \textbf{continuous, $2\pi$-periodic} function of $\omega\in\R$.
\end{defn}

\begin{defn}[title=Inverse DTFT (synthesis)]
\[
  x[n] = \frac{1}{2\pi}\int_{-\pi}^{\pi} X(\omega)\,e^{j\omega n}\,d\omega
\]
Uses $\int_{-\pi}^{\pi} e^{j\omega(n-k)}d\omega = 2\pi\,\delta[n-k]$.
\end{defn}

\subsection{Existence \& Convergence}

\begin{thm}[title=Existence: absolutely summable $\Rightarrow$ DTFT]
If $\sum_{n=-\infty}^{\infty}|x[n]|<\infty$, then $X(\omega)$ exists, is continuous, and $|X_N(\omega)|\to X(\omega)$ uniformly.
\end{thm}

\begin{prop}[title=Square-summable (finite energy)]
If $\mathbf{x}\in\ell_2(\mathbb{Z})$: $\sum|x[n]|^2<\infty$, DTFT exists in mean-square sense:
\[
  \lim_{N\to\infty}\int_{-\pi}^{\pi}|X_N(\omega)-X(\omega)|^2\,d\omega = 0
\]
$X(\omega)$ may not be continuous (Gibbs phenomenon possible).
\end{prop}

\subsection{DTFT as Limit of DFT}
As $N\to\infty$: $\omega_N = 2\pi/N \to 0$, DFT samples become dense over $[0,2\pi]$. By substituting $\omega = 2\pi k/N$:
\[
  X[k] \to X\!\left(\frac{2\pi k}{N}\right)
\]
Also: $\tilde{X}_N[k] = X\!\left(\frac{2\pi}{N}k\right)$ (DFS as samples of DTFT).

\section{DTFT Pairs \& Properties}

\subsection{Elementary DTFT Pairs}

\begin{prop}[title=Impulse]
$\mathrm{DTFT}\{\boldsymbol{\delta}_m\} = e^{-j\omega m}$\\
Most concentrated in time $\Rightarrow$ spread uniformly in freq.
\end{prop}

\begin{prop}[title=Rectangular signal]
$x[n]=\begin{cases}1 & 0\le n < M\\0 & \text{otherwise}\end{cases}$
\[
  X(\omega) = \frac{\sin(\omega M/2)}{\sin(\omega/2)}\,e^{-j\omega(M-1)/2}
\]
Zeros at $\omega = 2\pi m/M$, $m\ne 0$.
\end{prop}

\begin{prop}[title=Exponential decay ($|a|<1$)]
$x[n] = a^n u[n]$ ($u[n]$: unit step)
\[
  X(\omega) = \frac{1}{1-a\,e^{-j\omega}}
\]
$|X(\omega)|^2 = \dfrac{1}{1+|a|^2-2a\cos\omega}$
\end{prop}

\subsection{DTFT Properties}

\begin{prop}[title=Linearity]
$\mathrm{DTFT}\{a\mathbf{x}+b\mathbf{y}\} = a\mathbf{X}+b\mathbf{Y}$, $a,b\in\mathbb{C}$
\end{prop}

\begin{prop}[title=Time shift]
$x[n-n_0] \overset{\mathrm{DTFT}}{\longleftrightarrow} e^{-j\omega n_0}\,X(\omega)$
\end{prop}

\begin{prop}[title=Frequency shift (modulation)]
$x[n]\,e^{j\omega_0 n} \overset{\mathrm{DTFT}}{\longleftrightarrow} X(\omega - \omega_0)$
\[
  x[n]\cos(\omega_0 n)\overset{\mathrm{DTFT}}{\longleftrightarrow}\tfrac{1}{2}[X(\omega-\omega_0)+X(\omega+\omega_0)]
\]
\end{prop}

\begin{prop}[title=Time \& frequency reversal]
$\mathrm{DTFT}\{\mathcal{R}\mathbf{x}\} = \mathcal{R}\mathbf{X}$\\
$\mathrm{DTFT}\{\mathbf{x}^*\} = \mathcal{R}\mathbf{X}^*$
\end{prop}

\subsection{Symmetries of the DTFT}
A signal is \textbf{even} if $x[n]=x[-n]$, \textbf{odd} if $x[n]=-x[-n]$.\\
A signal is \textbf{Hermitian-symmetric} if $x[n]=x^*[-n]$.

\begin{thm}[title=Symmetry summary]
\renewcommand{\arraystretch}{1}
\begin{tabularx}{\linewidth}{@{}XX@{}}
\toprule
\textbf{Time domain} $x[n]$ & \textbf{Freq.\ domain} $X(\omega)$ \\
\midrule
Real & $X(-\omega)=X^*(\omega)$ \\
Real \& even & Real \& even \\
Real \& odd & Imaginary \& odd \\
Complex \& even & Complex \& even \\
Complex \& odd & Complex \& odd \\
\bottomrule
\end{tabularx}
\end{thm}

\begin{thm}[title=Parseval's theorem (DTFT)]
\[
  \sum_{n=-\infty}^{\infty}|x[n]|^2 = \frac{1}{2\pi}\int_{-\pi}^{\pi}|X(\omega)|^2\,d\omega
\]
\end{thm}

\subsection{Reading DTFT plots}
\begin{defn}[title=DTFT frequency axis]
Plotted over $[-\pi,\pi]$:
\begin{itemize}
  \item $\omega=0$: DC
  \item $\omega\in(0,\pi]$: positive (CCW) = high freqs near $\pm\pi$
  \item $\omega\in[-\pi,0)$: negative (CW) = high freqs near $\pm\pi$
  \item Low freqs near 0, high freqs near $\pm\pi$
\end{itemize}
DTFT is $2\pi$-periodic; plots outside $[-\pi,\pi]$ repeat.
\end{defn}

\subsection{Generalized DTFT (Power Signals)}
Via functionals for infinite-energy signals:
\begin{prop}[title=Constant \& complex exponential]
$\mathrm{DTFT}\{1\}=\delta(\omega)$ \quad (Dirac delta at $\omega=0$)\\
$e^{j\omega_0 n}\overset{\mathrm{DTFT}}{\longleftrightarrow}\delta(\omega-\omega_0)$
\end{prop}
\begin{prop}[title=Sinusoids]
$\cos(\omega_0 n)\overset{\mathrm{DTFT}}{\longleftrightarrow}\tfrac{1}{2}[\delta(\omega-\omega_0)+\delta(\omega+\omega_0)]$\\
$\sin(\omega_0 n)\overset{\mathrm{DTFT}}{\longleftrightarrow}\tfrac{-j}{2}[\delta(\omega-\omega_0)-\delta(\omega+\omega_0)]$
\end{prop}
\begin{prop}[title=Unit step]
$U(\omega)=\dfrac{1}{1-e^{-j\omega}}+\tfrac{1}{2}\delta(\omega)$
\end{prop}

\section{DFT vs.\ DTFT vs.\ DFS}

\renewcommand{\arraystretch}{1.15}
\begin{tabularx}{\linewidth}{@{}lXX@{}}
\toprule
 & Input & Output \\
\midrule
DFT  & finite, length $N$ & vector $\in\mathbb{C}^N$ \\
DFS  & $N$-periodic & $N$-periodic \\
DTFT & infinite ($\ell_2$) & $2\pi$-periodic function \\
\bottomrule
\end{tabularx}

\begin{warn}[title=DFT/DFS equivalence]
DFS of an $N$-periodic signal $=$ DFT of one period. Both use \textbf{identical formulas} (4.10) and (4.11); only the interpretation of periodicity differs.
\end{warn}

\begin{prop}[title=DTFT as DFS limit]
$\tilde{x}_N[n] = \sum_{p=-\infty}^{\infty} x[n+pN]$ (periodization).\\
$\tilde{X}_N[k] = X\!\left(\frac{2\pi}{N}k\right)$: DFS samples the DTFT.\\
As $N\to\infty$: $\tilde{x}_N\to\mathbf{x}$ and samples fill $[0,2\pi]$.
\end{prop}

\section{The FFT}

\begin{prop}[title=Complexity comparison]
\begin{itemize}
  \item Naive DFT: $O(N^2)$ multiplications
  \item FFT: $O(N\log N)$ operations
  \item For $N=10{,}000$: $\sim\!40{,}000$ vs.\ $>100$ million ops
\end{itemize}
\end{prop}

\begin{defn}[title=Decimation in time (Cooley-Tukey)]
Split $N=2^L$ input into even/odd:
$x_e[n]=x[2n]$, $x_o[n]=x[2n+1]$, $n=0,\ldots,\tfrac{N}{2}-1$

Output halves:
$X_a[k]=X[k]$ and $X_b[k]=X[k+\tfrac{N}{2}]$, $k=0,\ldots,\tfrac{N}{2}-1$

\textbf{Butterfly equations:}
\[
  X_a[k] = X_e[k] + e^{j\frac{2\pi}{N}k}\,X_o[k]
\]
\[
  X_b[k] = X_e[k] - e^{j\frac{2\pi}{N}k}\,X_o[k]
\]
where $X_e$, $X_o$ are the $(N/2)$-point DFTs of $x_e$, $x_o$.
\end{defn}

\begin{warn}[title=Cost breakdown]
\begin{itemize}
  \item $2(N/2)^2 = N^2/2$ mults for $\mathbf{X}_e$ and $\mathbf{X}_o$
  \item $N/2$ mults by twiddle factors for $\mathbf{X}_o$
  \item Second half is free (just sign change)
\end{itemize}
Recursion gives $O(N\log_2 N)$ total.
\end{warn}

\section{Spectrograms (STFT)}

\begin{defn}[title=Short-Time Fourier Transform]
Analyse a signal $\mathbf{x}$ of length $L$ time-locally:
\begin{enumerate}
  \item Slide a window of size $W$ to get $\lfloor L/W \rfloor$ non-overlapping bins
  \item Compute a $W$-point DFT in each bin
  \item Result: 2D time $\times$ frequency representation
\end{enumerate}
\end{defn}

\begin{prop}[title=Kronecker delta in spectrogram]
DFT of a length-$W$ Kronecker delta $\delta[n-m]$ (even shifted): $|X[k]|=1$ for all $k$.\\
$\Rightarrow$ In a spectrogram with window $W$: the bin containing $m$ has \textbf{constant magnitude} $W$ across all frequencies; all other bins are $\mathbf{0}$.
\end{prop}

\begin{warn}[title=Window size trade-off]
\textbf{Large} $W$: good frequency resolution, poor time resolution.\\
\textbf{Small} $W$: good time resolution, poor frequency resolution.\\
This is an instance of the time--frequency concentration duality.
\end{warn}

\section{Discrete-Time Filters}

\subsection{LTI Systems}

\begin{defn}[title=Linearity \& time-invariance]
Let $y[n] = \mathcal{H}\{x[n]\}$. \\
\textbf{Linearity} ($\forall\, \alpha,\beta \in\mathbb{C}$, $\forall\, x_1, x_2$):
\[
    \mathcal{H}\{\alpha\, x_1[n] + \beta\, x_2[n]\} \stackrel{?}{=} \alpha\, \mathcal{H}\{x_1[n]\} + \beta\, \mathcal{H}\{x_2[n]\}
\]

\textbf{Time-Invariance} ($\forall\, n_0 \in \mathbb{Z},\ \forall\, x$):
\[
    \underbrace{\mathcal{H}\{x[n - n_0]\}}_{x[\cdot] \,\leftarrow\, x[\cdot - n_0]} \stackrel{?}{=} \underbrace{y[n - n_0]}_{n \,\leftarrow\, n - n_0}
\]
\emph{Key consequence:} an LTI system cannot create new frequency content.

\end{defn}

\begin{defn}[title=Zero initial conditions]
A recursive filter is LTI and has a \emph{unique} output only if $\exists\, n_0$ such that for $n<n_0$: input is zero, output is zero, and all memory cells are zero. Nonzero initial memory breaks both uniqueness and time-invariance.
\end{defn}

\subsection{Impulse Response \& Convolution}

\begin{defn}[title=Impulse response]
$\mathbf{h}=\mathcal{H}\boldsymbol{\delta}$. An LTI system is \emph{fully} described by $\mathbf{h}$.
\end{defn}

\begin{defn}[title={Convolution $\mathbf{y}=\mathbf{x}*\mathbf{h}$}]
\[y[n]=\sum_{k=-\infty}^{\infty}x[k]\,h[n-k]\]
Well-defined when one sequence is abs.\ summable and the other is bounded (or both abs.\ summable).\\[2pt]
\textbf{To compute $y[n]$ sample by sample:}
time-reverse $h$, slide it to position $n$, multiply element-wise with $x$, sum.
\end{defn}

\begin{prop}[title=Convolution properties]
\textbf{Commutativity:} $\mathbf{x}*\mathbf{h}=\mathbf{h}*\mathbf{x}$\\
\textbf{Linearity:} $\mathbf{x}*(\alpha\mathbf{y}+\beta\mathbf{w})=\alpha(\mathbf{x}*\mathbf{y})+\beta(\mathbf{x}*\mathbf{w})$\\
\textbf{Time-shift:} $\mathbf{x}*(\mathcal{S}^k\mathbf{y})=\mathcal{S}^k(\mathbf{x}*\mathbf{y})$\\
\textbf{Associativity:} $(\mathbf{x}*\mathbf{h})*\mathbf{w}=\mathbf{x}*(\mathbf{h}*\mathbf{w})$ \emph{(abs.\ summable)}\\
\textbf{Cascade:} filters in series $\equiv$ single filter with $\mathbf{h}=\mathbf{h}_1*\mathbf{h}_2$; order is irrelevant.
\end{prop}

% --- 3. FILTER PROPERTIES FROM h ---
\subsection{\texorpdfstring{Filter Properties (from $\mathbf{h}$)}{Filter Properties (from h)}}

\begin{defn}[title=FIR vs IIR]
\begin{itemize}
  \item \textbf{FIR} (Finite Impulse Response): $\mathbf{h}$ has finite support. No feedback; \emph{always} BIBO stable.
  \item \textbf{IIR} (Infinite Impulse Response): $\mathbf{h}$ has infinite support. Realizable IIRs use feedback and have a one-sided $\mathbf{h}$.
\end{itemize}
\end{defn}

\begin{defn}[title=Causality]
Output at time $n$ depends only on $x[n], x[n-1],\ldots$ (past/present). Equivalently, $h[n]=0$ for $n<n_0$ with some $n_0\ge 0$.
\begin{itemize}
  \item \textbf{Causal up to a delay}: $h[n]=0$ for $n<n_0<0$ — FIR filters always satisfy this.
  \item \textbf{Anticausal}: $h[n]=0$ for $n>0$; offline use only.
  \item \textbf{Ideal}: $h$ has infinite two-sided support — \emph{not realizable}, used as theoretical benchmark.
\end{itemize}
\end{defn}

\begin{thm}[title=BIBO Stability]
LTI is BIBO stable $\Longleftrightarrow$ $\mathbf{h}$ is absolutely summable:
\[\sum_{k=-\infty}^{\infty}|h[k]|<\infty\]
\textbf{($\Leftarrow$)} If $|x[n]|\le L$ then $|y[n]| \le L\sum_k|h[k]| < \infty$. \textbf{($\Rightarrow$)} If $\sum|h[k]|=\infty$, take $x[n]=\mathrm{sgn}(h[-n])$; then $y[0]=\infty$.\\
\textbf{FIR is always BIBO stable} (finite sum is always finite).
\end{thm}

\subsection{Realizable Filters: CCDE}

\begin{defn}[title=Constant-Coefficient Difference Equation]
\[y[n]=\underbrace{\sum_{k=0}^{M-1}b_k\,x[n-k]}_{\text{feedforward (FIR part)}}-\underbrace{\sum_{k=1}^{N-1}a_k\,y[n-k]}_{\text{feedback (IIR part)}}\]
\textbf{FIR}: all $a_k=0$ for $k\ge 1$.\\
\textbf{IIR}: at least one $a_k\ne 0$ for $k\ge 1$ — recursion introduces a feedback loop.
\end{defn}

% --- 5. FREQUENCY RESPONSE ---
\subsection{Frequency Response}

\begin{thm}[title=Eigensequence property]
Complex exponentials are eigenfunctions of any LTI system:
\[\mathcal{H}\{e^{j\omega_0 n}\} = H(\omega_0)\cdot e^{j\omega_0 n}\]
\[H(\omega)=\sum_{k=-\infty}^{\infty}h[k]\,e^{-j\omega k}\]
$|H(\omega_0)|$ scales the amplitude; $\angle H(\omega_0)$ shifts the phase. $H(\omega)$ is the DTFT of $\mathbf{h}$.
\end{thm}

\begin{thm}[title=Convolution theorem]
\[y=x*h \;\Longleftrightarrow\; Y(\omega)=X(\omega)\cdot H(\omega)\]
Filtering in time $=$ pointwise multiplication in frequency. $H(\omega)$ acts as a \emph{spectral mask}.
\end{thm}

\begin{defn}[title=Filter types by magnitude response]
\begin{itemize}
  \item \textbf{Lowpass}: passes low freqs ($\omega\approx 0$), blocks high ($\omega\approx\pm\pi$)
  \item \textbf{Highpass}: passes high freqs, blocks low
  \item \textbf{Bandpass}: passes freqs near $\pm\omega_c$
  \item \textbf{Allpass}: $|H(\omega)|=\text{const}$ — modifies phase only
\end{itemize}
\textbf{Passband}: $|H|\approx 1$;\quad \textbf{Stopband}: $|H|\approx 0$.
\end{defn}

\begin{defn}[title=Linear phase \& group delay]
\textbf{Linear phase:} $\angle H(\omega)=-\omega d$ — all frequencies delayed by exactly $d$ samples. Generalized: $H(\omega)=A(\omega)e^{-j(\omega d+\phi)}$, $A(\omega)\in\mathbb{R}$.\\[2pt]
\textbf{Group delay:} $\mathrm{grd}\{H(\omega)\}=-\dfrac{d\,\angle H(\omega)}{d\omega}$

Constant group delay $\Leftrightarrow$ linear phase. All realizable linear-phase filters are FIR.
\end{defn}


% --- 6. EXAMPLES: MA and LI side by side ---
\subsection{Running Examples}

\begin{defn}[title=Moving Average (MA)]
\[y[n]=\frac{1}{M}\sum_{k=0}^{M-1}x[n-k],\quad h[n]=\begin{cases}\tfrac{1}{M}&0\le n<M\\0&\text{else}\end{cases}\]
Causal FIR $\Rightarrow$ always BIBO stable. Cost: $M{-}1$ additions $+$ 1 mult per sample.\\
\textbf{Zero-centered} (noncausal): $y[n]=\frac{1}{2L+1}\sum_{k=-L}^{L}x[n-k]$ — no lag; causal up to delay $L$.
\end{defn}

\begin{prop}[title=MA frequency response]
\[H(\omega)=\frac{1}{M}\,\frac{\sin(\omega M/2)}{\sin(\omega/2)}\,e^{-j\frac{M-1}{2}\omega}\]
Lowpass; $M{-}1$ nulls at $\omega=2\pi m/M$ ($m\ne 0$). Generalized linear phase, delay $\frac{M-1}{2}$.\\
Larger $M$ $\Rightarrow$ narrower passband, stronger smoothing.
\end{prop}

\begin{defn}[title=Leaky Integrator (LI)]
\[y[n]=\lambda\,y[n-1]+(1-\lambda)\,x[n], \quad |\lambda|<1\]
\[h[n]=(1-\lambda)\lambda^n\,u[n]\quad\text{(causal, exponential decay)}\]
Causal IIR. Constant cost: 2 mult $+$ 1 add per sample regardless of smoothing power.\\
Approximates MA of length $M$ with $\lambda=\frac{M-1}{M}$.
\end{defn}

\begin{prop}[title=LI frequency response \& stability]
\[H(\omega)=\frac{1-\lambda}{1-\lambda e^{-j\omega}}, \quad |H(\omega)|^2=\frac{(1-\lambda)^2}{1+\lambda^2-2\lambda\cos\omega}\]
Lowpass; bandwidth $\downarrow$ as $\lambda\to 1$. BIBO stable $\Leftrightarrow$ $|\lambda|<1$:
\[\textstyle\sum_{n\ge 0}|h[n]| = \tfrac{|1-\lambda|}{1-|\lambda|} < \infty \;\Leftrightarrow\; |\lambda|<1\]
At $\lambda=1$: perfect integrator — output diverges.
\end{prop}

% --- 7. IDEAL FILTERS ---
\subsection{Ideal Filters}

\begin{defn}[title=sinc \& rect]
$\mathrm{sinc}(x)=\dfrac{\sin(\pi x)}{\pi x}$,\; $\mathrm{sinc}(0)=1$.\qquad
$\mathrm{rect}(x)=\begin{cases}1&|x|\le\tfrac{1}{2}\\0&|x|>\tfrac{1}{2}\end{cases}$

DTFT pair: $\dfrac{\omega_p}{\pi}\,\mathrm{sinc}\!\left(\dfrac{\omega_p}{\pi}n\right)\overset{\mathrm{DTFT}}{\longleftrightarrow}\mathrm{rect}\!\left(\dfrac{\omega}{2\omega_p}\right)$
\end{defn}


\begin{prop}[title=Fundamental sinc/rect DTFT pair]
\[\frac{\omega_p}{\pi}\,\mathrm{sinc}\!\left(\frac{\omega_p}{\pi}n\right)\overset{\mathrm{DTFT}}{\longleftrightarrow}\mathrm{rect}\!\left(\frac{\omega}{2\omega_p}\right)\]
As $\omega_p\to\pi$: recovers $\boldsymbol{\delta}\leftrightarrow\mathbf{1}$.\quad As $\omega_p\to 0$: $\mathbf{1}\leftrightarrow\tilde{\delta}(\omega)$.
\end{prop}

\begin{prop}[title=Ideal Lowpass (cutoff $\omega_p$)]
$H_L(\omega)=\begin{cases}1&|\omega|\le\omega_p\\0&\text{else}\end{cases}$,\quad
$h_L[n]=\dfrac{\omega_p}{\pi}\,\mathrm{sinc}\!\left(\dfrac{\omega_p}{\pi}n\right)$\\
Infinite two-sided support $\Rightarrow$ \textbf{not realizable}.
\end{prop}

\begin{prop}[title=Ideal Highpass \& Bandpass]
$H_H(\omega)=1-H_L(\omega)$,\quad $h_H[n]=\delta[n]-h_L[n]$\\[2pt]
$h_B[n]=2\cos(\omega_c n)\,\dfrac{\omega_p}{\pi}\,\mathrm{sinc}\!\left(\dfrac{\omega_p}{\pi}n\right)$ (BW $2\omega_p$, center $\omega_c$)
\end{prop}

\begin{prop}[title=Hilbert Filter (allpass)]
$H_F(\omega)=\begin{cases}-j&0\le\omega<\pi\\+j&-\pi\le\omega<0\end{cases}$,\quad $|H_F|=1$\\
$h_F[n]=\begin{cases}2/(n\pi)&n\text{ odd}\\0&n\text{ even}\end{cases}$
\end{prop}

\section{Z-Transform \& Filter Design}

% --- 1. Z-TRANSFORM ---
\subsection{Z-Transform}

\begin{defn}[title=Z-Transform]
\[\mathcal{Z}\{\mathbf{x}\}(z) = X(z) = \sum_{n=-\infty}^{\infty} x[n]\,z^{-n}, \quad z\in\mathbb{C}\]
A formal power series in $z^{-1}$; existence requires the sum to converge absolutely for a given $z$.
\end{defn}

\begin{prop}[title=Fundamental properties]
\textbf{Linearity:} $\mathcal{Z}\{\alpha\mathbf{x}+\beta\mathbf{y}\} = \alpha X(z)+\beta Y(z)$\\
\textbf{Time-shift:} $\mathcal{Z}\{x[n-N]\} = z^{-N}X(z)$ \quad (delay by $N$ multiplies by $z^{-N}$)\\
\textbf{Convolution:} $\mathcal{Z}\{\mathbf{h}*\mathbf{x}\} = H(z)\cdot X(z)$
\end{prop}

\begin{prop}[title=Link to DTFT]
Evaluating on the unit circle $z=e^{j\omega}$ recovers the DTFT:
\[X(e^{j\omega}) = \sum_{n=-\infty}^{\infty} x[n]\,e^{-j\omega n} = \mathrm{DTFT}\{\mathbf{x}\}\]
The DTFT is the z-transform restricted to $|z|=1$ (when the unit circle is inside the ROC).
\end{prop}

% --- 2. TRANSFER FUNCTION ---
\subsection{Transfer Function}

\begin{defn}[title=Transfer function $H(z)$]
The z-transform of the impulse response $\mathbf{h}$:
\[H(z) = \mathcal{Z}\{\mathbf{h}\} = \frac{Y(z)}{X(z)}\]
For a CCDE (apply z-transform to both sides):
\[H(z) = \frac{\displaystyle\sum_{k=0}^{M}b_k z^{-k}}{\displaystyle 1+\sum_{k=1}^{N}a_k z^{-k}} = \frac{B(z)}{A(z)}\]
$B(z)$: feedforward (FIR part). $A(z)$: feedback (IIR part). FIR: $A(z)=1$.
\end{defn}

\begin{defn}[title=Zeros \& poles — factored form]
\[H(z) = b_0\,\frac{\displaystyle\prod_{k=0}^{M-1}(1-z_k z^{-1})}{\displaystyle\prod_{k=0}^{N-1}(1-p_k z^{-1})}\]
\textbf{Zeros} $z_k$: roots of $B(z)$, where $|H|=0$.\\
\textbf{Poles} $p_k$: roots of $A(z)$, where $|H|\to\infty$.\\
Coinciding pole and zero cancel (reduce order).
\end{defn}

% --- 3. REGION OF CONVERGENCE ---
\subsection{Region of Convergence (ROC)}

\begin{defn}[title=ROC definition]
The set of $z\in\mathbb{C}$ for which $X(z)$ converges absolutely. Always circularly symmetric (depends only on $|z|$).
\end{defn}

\begin{prop}[title=ROC shapes]
\begin{itemize}
  \item \textbf{FIR} (finite support): entire $\mathbb{C}$, possibly except $z=0$ or $z=\infty$.
  \item \textbf{Causal IIR}: exterior of disk, $|z|>R$, where $R=\max_k|p_k|$.
  \item \textbf{Anticausal IIR}: interior of disk, $|z|<r$, where $r=\min_k|p_k|$.
  \item \textbf{Two-sided}: annular region $r<|z|<R$. Can be empty.
\end{itemize}
\emph{ROC shape encodes causality.} Same $H(z)$ expression with different ROC = different causal nature.
\end{prop}

% --- 4. STABILITY VIA POLES ---
\subsection{Stability via Poles}

\begin{thm}[title=BIBO stability $\Leftrightarrow$ unit circle in ROC]
LTI is BIBO stable iff unit circle $|z|=1$ lies inside the ROC of $H(z)$:
\[1\in\mathrm{ROC}\{H(z)\} \Longleftrightarrow \sum_n |h[n]|<\infty\]
\textbf{For causal systems:} BIBO stable $\Longleftrightarrow$ \textbf{all poles inside the unit circle} ($|p_k|<1$).
\end{thm}

\begin{prop}[title=Pole-zero cancellation in cascades]
In $H(z)=H_1(z)\cdot H_2(z)$: a zero of $H_1$ at the same location as a pole of $H_2$ formally cancels. \textbf{Caution:} cancellation is never exact in finite precision — an unstable pole cannot be reliably eliminated this way.
\end{prop}

% --- 5. POLE-ZERO PLOT & MAGNITUDE RESPONSE ---
\subsection{Pole-Zero Plot \& Magnitude Response}

\begin{defn}[title=Pole-zero plot]
Plot poles ($\times$) and zeros ($\bullet$) on the complex plane. Unit circle $|z|=1$ as reference — frequency response read off along it ($\omega\in[0,\pi]$ = upper half).
\end{defn}

\begin{prop}[title=Symmetry patterns]
\textbf{Real-valued filter:} complex poles/zeros appear in conjugate pairs $(p, p^*)$. Plot symmetric about real axis.\\[2pt]
\textbf{Linear-phase FIR:} zeros appear in reciprocal pairs: if $z_0$ is a zero, so is $1/z_0^*$. Real coefficients: zero $z_0$ generates quadruple $\{z_0,\,z_0^*,\,1/z_0,\,1/z_0^*\}$.
\end{prop}

\begin{prop}[title={Sketching $|H(\omega)|$ from pole-zero plot}]
\begin{itemize}
  \item \textbf{Zero on unit circle} $\Rightarrow$ exact null at that $\omega$.
  \item \textbf{Pole near unit circle} $\Rightarrow$ $|H(\omega)|$ peaks nearby.
  \item \textbf{Zero near unit circle} $\Rightarrow$ $|H(\omega)|$ dips nearby.
\end{itemize}
\end{prop}

% --- 6. RUNNING EXAMPLES IN Z-DOMAIN ---
\subsection{Running Examples in Z-Domain}

\begin{prop}[title=Moving Average — z-domain]
\[H(z)=\frac{1}{N}\cdot\frac{1-z^{-N}}{1-z^{-1}}=\frac{1}{N}\prod_{k=1}^{N-1}(1-e^{j\frac{2\pi}{N}k}z^{-1})\]
$N{-}1$ zeros at $N$-th roots of unity (all except $z=1$); no poles. ROC: all $\mathbb{C}$ except $z=0$. Always stable (FIR).
\end{prop}

\begin{prop}[title=Leaky Integrator — z-domain]
\[H(z)=\frac{1-\lambda}{1-\lambda z^{-1}}, \quad \text{pole at } z=\lambda\]
ROC: $|z|>\lambda$ (causal). Stable $\Leftrightarrow$ $|\lambda|<1$ (pole inside unit circle). As $\lambda\to 1$: pole approaches unit circle, bandwidth narrows.
\end{prop}

% --- 7. FILTER REALIZATIONS ---
\subsection{Filter Realizations}

\begin{defn}[title=Direct Form I \& II]
\textbf{Direct Form I}: implement CCDE directly — separate delay lines for $x$ and $y$; $M+N$ total delays.\\[2pt]
\textbf{Direct Form II}: shared single delay line, $\max(M,N)$ elements. Same transfer function, fewer memory cells.\\[2pt]
\textbf{Second-order sections (SOS / biquads)}: factor $H(z)=b_0\prod_k H_k(z)$ into 2nd-order real-coefficient sections, each as Direct Form II. Prevents pole drift in finite-precision arithmetic — \emph{required for high-order IIRs}. (\texttt{output='sos'} in SciPy.)
\end{defn}

% --- 8. INTUITIVE DESIGN (POLE-ZERO PLACEMENT) ---
\subsection{Intuitive Design (Pole-Zero Placement)}

\begin{prop}[title=Design principle]
Place \textbf{zeros on the unit circle} to create exact nulls. Place \textbf{poles near the unit circle} to boost gain. Response at $\omega$ = product of distances from zeros / product of distances from poles.
\end{prop}

\begin{prop}[title=DC notch \& hum removal]
\textbf{DC removal}: $H(z)=1-z^{-1}$ — zero at $z=1$ ($\omega=0$), but over-attenuates nearby freqs.\\[2pt]
\textbf{DC notch} (improved): add compensating pole near $z=1$:
\[H(z)=\frac{1-z^{-1}}{1-rz^{-1}},\quad r\lesssim 1\]
As $r\to 1$: notch narrows. Stable iff $r<1$.\\[2pt]
\textbf{Hum removal}: zeros at $e^{\pm j\omega_0}$ with $\omega_0=2\pi f_0/f_s$. Add poles nearby at $re^{\pm j\omega_0}$ to control notch width.
\end{prop}

\begin{prop}[title=Resonator]
\textbf{Simple resonator} — poles at $z=re^{\pm j\omega_0}$:
\[H(z)=\frac{1}{1-2r\cos\omega_0\cdot z^{-1}+r^2 z^{-2}}\]
Gain peaks at $\omega_0$; bandwidth narrows as $r\to 1$. Stable iff $r<1$.
\end{prop}

% --- 9. IIR FILTER FAMILIES ---
\subsection{IIR Filter Families}

\begin{defn}[title=dB scale]
$|H(\omega)|_{\mathrm{dB}}=20\log_{10}|H(\omega)|$ (reference: passband $= 0\,\mathrm{dB}$).\\
$-6\,\mathrm{dB}$: half amplitude;\quad $-20\,\mathrm{dB}$: one-tenth amplitude;\quad $A_s=-20\log_{10}\delta_s$.
\end{defn}

\begin{prop}[title=Classic IIR lowpass designs]
\textbf{Butterworth}: all-pole, monotone (maximally flat) in both bands. Best phase linearity; gradual roll-off. Design: order $N$, cutoff $\omega_c$ at $-3\,\mathrm{dB}$.\\[2pt]
\textbf{Chebyshev}: equiripple in passband, monotone in stopband. Narrower transition band than Butterworth for same $N$.\\[2pt]
\textbf{Elliptic}: equiripple in \emph{both} bands — most order-efficient (narrowest transition band for given $N$). Design: $N$, $\omega_c$, $\delta_p$, $\delta_s$.
\end{prop}

\begin{prop}[title={IIR comparison (same order $N$)}]
\textbf{Narrowest trans.\ band}: Elliptic $>$ Chebyshev $>$ Butterworth\\
\textbf{Passband flatness}: Butterworth $>$ Chebyshev $>$ Elliptic\\
\textbf{Phase linearity}: Butterworth $>$ Chebyshev $>$ Elliptic
\end{prop}

\begin{prop}[title=IIR vs FIR]
\textbf{IIR} pros: efficient (low order); strong attenuation\\
\textbf{IIR} cons: stability issues; always nonlinear phase\\
\textbf{FIR} pros: always stable; numerically robust; linear phase\\
\textbf{FIR} cons: high order (long delay); no analog equivalent
\end{prop}

% --- 10. FIR FILTER DESIGN ---
\subsection{FIR Filter Design}

\begin{defn}[title=Filter specifications]
\textbf{Passband} $|\omega|\le\omega_p$: $1-\delta_p\le|H|\le 1+\delta_p$\quad
\textbf{Stopband} $|\omega|\ge\omega_s$: $|H|\le\delta_s$\\
\textbf{Transition band} $[\omega_p,\omega_s]$: unconstrained; $\Delta\omega=\omega_s-\omega_p$.\quad $A_s=-20\log_{10}\delta_s$ (dB).
\end{defn}

\begin{defn}[title=Windowing method]
Truncate ideal (infinite, non-causal) impulse response $h_d[n]$ with a length-$M$ window:
\[h[n]=h_d\!\left[n-\tfrac{M-1}{2}\right]\cdot w[n],\quad n=0,\ldots,M-1\]
Shifting by $\frac{M-1}{2}$ makes the filter causal with linear phase (group delay $\frac{M-1}{2}$). In frequency: $H=H_d*W$ — ideal cutoff smeared by window's main lobe.\\
\textbf{Gibbs phenomenon}: rectangular window gives $\approx\!9\%$ ripple near cutoff regardless of $M$. Tapering windows reduce ripple at the cost of wider transition band.
\end{defn}

\begin{prop}[title={Window functions (length $M$)}]
\begin{tabular}{@{}l@{~~}r@{~~}r@{}}
\textbf{Window} & \textbf{Atten.} & \textbf{Trans.\ BW}\\
\hline
Rectangular & 13 dB & $4\pi/M$\\
Hanning     & 44 dB & $8\pi/M$\\
Hamming     & 41 dB & $8\pi/M$\\
Blackman    & 74 dB & $12\pi/M$\\
\end{tabular}\\[2pt]
\textbf{Hanning:} $w[n]=\tfrac{1}{2}(1-\cos\tfrac{2\pi n}{M-1})$\quad
\textbf{Hamming:} $w[n]=0.54-0.46\cos\tfrac{2\pi n}{M-1}$\\
\textbf{Blackman:} $w[n]=0.42-0.5\cos\tfrac{2\pi n}{M-1}+0.08\cos\tfrac{4\pi n}{M-1}$
\end{prop}

\begin{prop}[title=Windowing design procedure (lowpass)]
\begin{enumerate}
  \item Specify $\omega_p$, $\omega_s$, $A_s$ (dB). Compute $\Delta\omega=\omega_s-\omega_p$.
  \item Choose window with attenuation $\ge A_s$; note its BW formula $C\pi/M$.
  \item Length: $M\approx C\pi/\Delta\omega$ (round up to odd for Type I).
  \item Ideal cutoff at midpoint: $\omega_c=(\omega_p+\omega_s)/2$.
  \item Ideal response: $h_d[n]=\tfrac{\omega_c}{\pi}\mathrm{sinc}(\tfrac{\omega_c}{\pi}n)$.
  \item Apply: $h[n]=h_d[n-\tfrac{M-1}{2}]\cdot w[n]$.
\end{enumerate}
HP: $h_H[n]=(\delta[n]-h_d[n-\tfrac{M-1}{2}])\cdot w[n]$.\\
BP: $h_B[n]=2\cos(\omega_c n)\,h_d[n-\tfrac{M-1}{2}]\cdot w[n]$.
\end{prop}

\begin{prop}[title={Linear-phase FIR types}]
Linear phase requires symmetric (S) or antisymmetric (A) coefficients. Delay $C=\tfrac{M-1}{2}$.\\[2pt]
\begin{tabular}{@{}c@{~~}c@{~~}c@{~~}c@{~~}l@{}}
\textbf{Type} & \textbf{$M$} & \textbf{Sym} & \textbf{Delay} & \textbf{Forced zeros at}\\
\hline
I   & odd  & S & integer  & none — most general\\
II  & even & S & half-int & $\omega=\pi$\\
III & odd  & A & integer  & $\omega=0$ and $\omega=\pm\pi$\\
IV  & even & A & half-int & $\omega=0$\\
\end{tabular}\\[2pt]
Type I/II: LP and HP ok. Type III: bandpass only (zeros at DC and Nyquist). Type IV: no LP.
\end{prop}

\begin{prop}[title=Parks-McClellan (optimal equiripple FIR)]
Minimizes the \emph{maximum} error (minimax, $L_\infty$ norm) in passband and stopband — error is \textbf{equiripple}. Contrast windowing: no control over max error.\\[2pt]
\textbf{Inputs}: filter length $M$, $\omega_p$, $\omega_s$, tolerance ratio $\delta_s/\delta_p$.\\
\textbf{Output}: $M$ optimal coefficients. Increase $M$ and retry if specs not met.\\
\emph{Result}: shorter filter than windowing for same specs; works for HP/BP/Hilbert/differentiator.
\end{prop}

\section{Quick Reference}

\subsection{Geometric sum}
\[
  \sum_{n=0}^{N-1}r^n = \frac{1-r^N}{1-r},\quad r\ne 1
\]
Used in rectangular DFT derivation; also the orthogonality of Fourier vectors (set $r=e^{j\frac{2\pi}{N}(k-h)}$, $k\ne h$).

\subsection{Euler \& trig}
\begin{itemize}

  \item $(e^{j\theta})^* = e^{-j\theta}$
  \item $\cos(a+b)=\cos a\cos b - \sin a\sin b$
  \item $\sin(a+b)=\sin a\cos b + \cos a\sin b$
  \item $\sin a\sin b = \tfrac{1}{2}[\cos(a-b)-\cos(a+b)]$
  \item $\cos a\cos b = \tfrac{1}{2}[\cos(a-b)+\cos(a+b)]$
  \item $\sin a\cos b = \tfrac{1}{2}[\sin(a+b)+\sin(a-b)]$
\end{itemize}

\subsection{Complex number tricks}
\begin{itemize}
  \item $z_1 = z_2^* \Rightarrow |z_1|=|z_2|$ \quad ($|z|=|z^*|$ always)
  \item $z + z^* = 2\,\mathrm{Re}\{z\}$, \quad $z - z^* = 2j\,\mathrm{Im}\{z\}$
  \item $z = z^* \;\Longleftrightarrow\; z\in\mathbb{R} \;\Longleftrightarrow\; \mathrm{Im}\{z\}=0$
  \item $|z_1 z_2| = |z_1||z_2|$, \quad $\angle(z_1 z_2) = \angle z_1 + \angle z_2$
  \item $|z|^2 = z\cdot z^*$
\end{itemize}

\subsection{\texorpdfstring{Inner product in $\mathbb{C}^N$}{Inner product in CN}}
Coefficients: $X[k]=\langle\mathbf{w}_k,\mathbf{x}\rangle$ (projection onto basis).

\subsection{{DTFT inner product in \texorpdfstring{$L_2([-\pi,\pi])$}{L2([-pi,pi])}}}
$\langle\mathbf{F},\mathbf{G}\rangle = \frac{1}{2\pi}\int_{-\pi}^{\pi}F^*(\omega)\,G(\omega)\,d\omega$\\
Set $\{e^{-j\omega n}\}_{n\in\Z}$ forms an orthogonal basis.\\
$x[n] = \langle e^{j\omega n}, X(\omega)\rangle$

\subsection{Key warnings}
\begin{warn}[title=DFT normalization]
\textbf{Analysis}: no $1/N$ factor. \textbf{Synthesis}: $1/N$ factor. Convention: $\mathbf{W}^H\mathbf{W}=N\mathbf{I}$.
\end{warn}

\begin{warn}[title=Phase wrapping]
Phase plots show $\angle X[k] \in [-\pi,\pi]$ (wrapped). Apparent discontinuities are $\pm\pi$ jumps, not real.
\end{warn}

\begin{warn}[title=Real signal DFT]
For real $\mathbf{x}$: $X[k]=X^*[N-k]$. Only first $\lfloor N/2\rfloor+1$ coefficients are independent. Magnitude symmetric around $k=N/2$.
\end{warn}

\begin{warn}[title=Circular vs.\ linear shift]
DFT/DFS time shifts are always \textbf{circular} (modulo $N$). For aperiodic finite signals, multiplying by phase in freq domain gives circular shift.
\end{warn}
