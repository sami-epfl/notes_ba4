\section{Flow network}
\subsection{Definitions}
\begin{center}
    \includegraphics[width=\linewidth]{images/flow_network.png}
\end{center}

\begin{defn}[title=Formally]
\begin{itemize}
    \item Source: $s\in V$, sink: $t\in V$
    \item Directed graph (DG): $G=(V,E)$
    \item Each (u, v) has a capacity $c(u,v) \ge 0$
    \item No antiparallel: if $(u,v)\in E$, then $(v,u)\not\in E$
\end{itemize}
\end{defn}

It satisfies the following properties:
\begin{thm}[title=Capacity constraint]
$\forall u,v\in V$, $0\le f(u,v)\le c(u,v)$
\end{thm}
\begin{thm}[title=Flow conservation]
$\forall u\in V\setminus\{s,t\}$,
$\sum_{v\in V} f(u,v) = \sum_{v\in V} f(v,u)$
\end{thm}

\begin{defn}[title=Cut]
A cut of a flow network is a partition of $V$ into two disjoint sets $S$ and $T$ such that $s\in S$ and $t\in T$. The capacity of the cut is defined as $c(S,T)=\sum_{u\in S, v\in T} c(u,v)$.
\end{defn}

\subsection{Maximum flow problem - Ford-Fulkerson method}

\begin{prop}[title={Residual network $G_f=(V,E_f)$}]
$\forall (u,v)\in E_f$, $c_f(u,v)$ is the residual capacity:
\begin{itemize}
    \item Additional flow possible (forward edge)
    \item Flow removable (backward edge)
\end{itemize}
\end{prop}

\begin{lstlisting}
FORD-FULKERSON-METHOD(G, s, t)
    initialize flow f to 0
    while exists an augmenting path p in G_f
        b = bottleneck(p)
        augment flow along p by b
    return f
\end{lstlisting}

\subsection{Min cut problem}

\begin{defn}[title=Min cut]
It is a cut where the capacity crossing it is the minimum among all cuts.
\end{defn}

\begin{prop}[title=Capacity of a cut]
$c(S,T) = \sum_{u\in S, v\in T} c(u,v)$
\end{prop}

\begin{prop}[title=Net flow accross cut]
$f(S,T) = \sum_{u\in S, v\in T} f(u,v) - \sum_{u\in S, v\in T} f(v,u)$
\end{prop}

\begin{thm}[title=Max-flow min-cut theorem]
\begin{itemize}
  \item f is the maximum flow
  \item $G_f$ has no augmenting path
  \item $|f| = c(S,T)$ for a minimum cut (S, T)
\end{itemize}
\end{thm}

\subsection{Applications of max flow}
\begin{prop}[title=Bipartite matching as flow problem]
Goal: find a maximum matching in a bipartite graph $G=(V,E)$ with $V=L\cup R$ (left and right parts).
\begin{itemize}
  \item Connect s to $\forall v\in L$ and $\forall v\in R$ to t
  \item $c(s,u) = 1$ for $u\in L$,
  \item $c(u,v) = 1$ for $(u,v)\in E$,
  \item $c(v,t) = 1$ for $v\in R$
  \item Run the Ford-Fulkerson method
  \item Matching is complete
\end{itemize}
\end{prop}

\begin{prop}[title=Edge-disjoint paths]
Given a directed graph $G=(V,E)$ and two vertices $s,t\in V$, the maximum number of edge-disjoint paths from $s$ to $t$ is equal to the minimum number of edges whose removal disconnects $s$ from $t$.
\begin{itemize}
  \item max-flow = max number of edge-disjoint paths
  \item min-cut = min number of edges to remove to disconnect $s$ from $t$
\end{itemize}
\end{prop}

\subsection{Key definiti2ons}
\begin{defn}[title=Linear map]
$T:V\to W$ is linear if $T(\alpha x+\beta y)=\alpha T(x)+\beta T(y)$
for all $x,y\in V$, $\alpha,\beta\in\R$.
\end{defn}

\begin{thm}[title=Rank--nullity]
For $T:V\to W$ with $\dim V<\infty$:
\[ \dim\ker T + \dim\operatorname{im}T \;=\; \dim V. \]
\end{thm}

\subsection{Formulas to remember}
\begin{itemize}
  \item \kv{Quadratic}{$x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$}
  \item \kv{Euler}{$e^{i\theta}=\cos\theta+i\sin\theta$}
  \item \kv{Geom.\ sum}{$\sum_{k=0}^{n}r^k=\dfrac{1-r^{n+1}}{1-r}$, $r\neq 1$}
  \item \kv{Taylor}{$f(x)=\sum_{k\ge0}\dfrac{f^{(k)}(a)}{k!}(x-a)^k$}
\end{itemize}

\begin{warn}[title=Pitfall]
$\log(ab)=\log a+\log b$ \emph{only} when $a,b>0$.
\end{warn}

 \cos(2\theta) &= \cos^2\theta-\sin^2\theta \\
 \sin(2\theta) &= 2\sin\theta\cos\theta
\end{align*}

\section{Topic 2 --- Probability}

\subsection{Basics}
\begin{itemize}
  \item $\PR(A\cup B)=\PR(A)+\PR(B)-\PR(A\cap B)$
  \item $\PR(A\mid B)=\dfrac{\PR(A\cap B)}{\PR(B)}$
  \item Bayes: $\PR(A\mid B)=\dfrac{\PR(B\mid A)\PR(A)}{\PR(B)}$
\end{itemize}

\subsection{Common distributions}
\renewcommand{\arraystretch}{1.05}
\begin{tabularx}{\linewidth}{@{}lXl@{}}
\toprule
Dist. & PMF/PDF & $\E[X]$ \\
\midrule
Bern$(p)$  & $p^x(1-p)^{1-x}$           & $p$ \\
Bin$(n,p)$ & $\binom{n}{x}p^x(1-p)^{n-x}$ & $np$ \\
Poiss$(\lambda)$ & $e^{-\lambda}\lambda^x/x!$ & $\lambda$ \\
$\mathcal{N}(\mu,\sigma^2)$ & $\tfrac{1}{\sigma\sqrt{2\pi}}e^{-(x-\mu)^2/2\sigma^2}$ & $\mu$ \\
\bottomrule
\end{tabularx}

\begin{ex}[title=Worked]
$X\sim \mathcal{N}(0,1)$. Then $\PR(|X|<1)\approx 0.6827$.
\end{ex}

\begin{lstlisting}[language=Python]
def bsearch(a, x):
    lo, hi = 0, len(a)-1
    while lo <= hi:
        m = (lo+hi)//2
        if a[m] == x: return m
        if a[m] <  x: lo = m+1
        else:         hi = m-1
    return -1
\end{lstlisting}
