\section{Core Concepts}
% ============================================================

\subsection{Empirical Risk}
\begin{defn}[title=Empirical Risk]
\[ R(w) = \frac{1}{N}\sum_{i=1}^{N} \ell(\hat{y}_i,\, y_i) \]
Training $\Leftrightarrow$ $\min_w R(w)$
\end{defn}

\subsection{Gradient Descent}
\begin{prop}[title=Gradient Descent Update]
\[ w_k \leftarrow w_{k-1} - \eta\,\nabla_w R(w_{k-1}) \]
$\eta$ = learning rate. Converges when $\nabla_w R(w)=0$.
\end{prop}

\begin{defn}[title=Convergence Criteria]
\begin{itemize}
  \item $|R(w_k)-R(w_{k-1})|<\delta_R$
  \item $\|w_k - w_{k-1}\| < \delta_w$
  \item or fixed number of iterations
\end{itemize}
\end{defn}

\subsection{Useful Identities}
\begin{itemize}
  \item $\nabla_b(a^T b)=a$
  \item $\|a\|^2 = a^T a$
  \item $\nabla_a\|a\|^2 = 2a$
\end{itemize}

% ============================================================
\section{Linear Regression}
% ============================================================

\subsection{Model \& Training}
\begin{defn}[title=Prediction]
\[ \hat{y}_i = w^T x_i \]
\end{defn}

\begin{defn}[title=MSE Risk]
\[ R(w) = \frac{1}{N}\sum_{i=1}^{N}(w^T x_i - y_i)^2 \]
\end{defn}

\begin{thm}[title=Closed-Form Solution]
\[ w^\star = (X^T X)^{-1} X^T y \]
If $X^TX$ not invertible: $w^\star = X^\dagger y$.\\
\textbf{Multi-output:} $W^\star = X^\dagger Y$
\end{thm}

\subsection{Evaluation Metrics}
\begin{itemize}
  \item \kv{MSE}{$\frac{1}{N}\sum(y_i-\hat{y}_i)^2$}
  \item \kv{RMSE}{$\frac{1}{N}\sum\sqrt{(y_i-\hat{y}_i)^2}$}
  \item \kv{MAE}{$\frac{1}{N}\sum|\hat{y}_i - y_i|$}
  \item \kv{MAPE}{$\frac{1}{N}\sum\left|\frac{\hat{y}_i - y_i}{y_i}\right|$}
\end{itemize}

% ============================================================
\section{Logistic Regression (Binary)}
% ============================================================

\subsection{Model}
\begin{defn}[title=Sigmoid]
\[ \sigma(z) = \frac{1}{1+e^{-z}}, \qquad \hat{y}_i = \sigma(w^T x_i) \]
\end{defn}

\subsection{Training}
\begin{defn}[title=Cross-Entropy Loss]
\[ R(w) = -\sum_{i=1}^{N}\!\Big[y_i\ln\hat{y}_i + (1-y_i)\ln(1-\hat{y}_i)\Big] \]
\end{defn}

\begin{prop}[title=Gradient]
\[ \nabla_w R(w) = \sum_{i=1}^{N}(\hat{y}_i - y_i)\,x_i \]
No closed form $\Rightarrow$ use \textbf{gradient descent}.
\end{prop}

\subsection{Prediction}
\[ \hat{y} = \begin{cases} 1 & \text{if } \sigma(w^{\star T}x) \geq 0.5 \\ 0 & \text{otherwise} \end{cases} \]

% ============================================================
\section{Classification Evaluation}
% ============================================================

\subsection{Confusion Matrix}
\renewcommand{\arraystretch}{1.1}
\begin{tabularx}{\linewidth}{@{}lXX@{}}
\toprule
 & \textbf{True Pos} & \textbf{True Neg} \\
\midrule
\textbf{Pred Pos} & TP & FP \\
\textbf{Pred Neg} & FN & TN \\
\bottomrule
\end{tabularx}

\vspace{2pt}
$P = TP+FN$, \quad $N = FP+TN$

\subsection{Metrics}
\begin{itemize}
  \item \kv{Accuracy}{$\frac{TP+TN}{TP+TN+FP+FN}$}
  \item \kv{Precision}{$\frac{TP}{TP+FP}$}
  \item \kv{Recall}{$\frac{TP}{TP+FN}$}
  \item \kv{FPR}{$\frac{FP}{FP+TN}$}
  \item \kv{F1}{$2\cdot\frac{\mathrm{Precision}\cdot\mathrm{Recall}}{\mathrm{Precision}+\mathrm{Recall}}$}
\end{itemize}

\begin{warn}[title=Multi-class]
Use \textbf{confusion matrix} (rows = predicted, cols = actual).
Accuracy = sum of diagonal / total.
Precision/Recall/F1: compute per class then average.
\end{warn}

% ============================================================
\section{Multi-Class Classification}
% ============================================================

\subsection{One-Hot Encoding}
$y_i \in \{0,1\}^C$: one entry is 1, rest are 0.

\subsection{Softmax (Logistic)}
\begin{defn}[title=Softmax]
\[ \hat{y}^{(k)}(x) = \frac{\exp(w^{(k)T}x)}{\sum_{j=1}^{C}\exp(w^{(j)T}x)} \]
\end{defn}

\begin{defn}[title=Multi-Class Cross-Entropy]
\[ R(W) = -\sum_{i=1}^{N}\sum_{k=1}^{C} y_i^{(k)}\ln\hat{y}_i^{(k)} \]
\end{defn}

\begin{prop}[title=Gradient]
\[ \nabla_W R(W) = \sum_{i=1}^{N} x_i\,(\hat{y}_i - y_i)^T \]
\end{prop}

\textbf{Prediction:} $k = \argmax_j\; \hat{y}^{(j)}$

% ============================================================
\section{k-Nearest Neighbors}
% ============================================================

\subsection{Distances}
\begin{itemize}
  \item \kv{Euclidean}{$\sqrt{\sum_{k}(x_i^{(k)}-x_j^{(k)})^2}$}
  \item \kv{Chi-square}{$\sqrt{\sum_{k}\frac{(x_i^{(k)}-x_j^{(k)})^2}{x_i^{(k)}+x_j^{(k)}}}$}
\end{itemize}

\subsection{Classification}
Find $k$ nearest neighbors $\Rightarrow$ majority vote.

\subsection{Regression}
Find $k$ nearest $\Rightarrow$ (weighted) average:
\[ \hat{y} = \frac{\sum_i w_i y_i}{\sum_i w_i}, \quad w_i = \frac{1}{d_i} \]

% ============================================================
\section{Feature Expansion \& Regularization}
% ============================================================

\subsection{Polynomial Feature Expansion}
\begin{defn}[title=Expansion]
$\phi: \R^D \to \R^F$ maps $x$ to higher-dim features.\\
$\hat{y}_i = w^T\phi(x_i)$
\end{defn}

\begin{thm}[title=Closed-Form with Expansion]
\[ w^\star = (\Phi^T\Phi)^{-1}\Phi^T y \]
\end{thm}

\begin{thm}[title=Cover's Theorem]
A complex classification problem is more likely linearly separable in high-dimensional space (if not densely populated).
\end{thm}

\subsection{Overfitting \& Underfitting}
\begin{itemize}
  \item \kv{Overfitting}{low train error, high test error}
  \item \kv{Underfitting}{high error on both}
\end{itemize}

\subsection{Cross-Validation}
\begin{itemize}
  \item \kv{Val.\ set}{80/20 split â€” cheap, wastes data}
  \item \kv{LOOCV}{train on $N{-}1$, test on 1, repeat â€” exact, expensive}
  \item \kv{K-fold}{split into $k$ parts, rotate â€” best compromise}
\end{itemize}

\subsection{L2 Regularization (Ridge)}
\begin{defn}[title=Regularized Objective]
\[ E(w) = R(w) + \lambda\,w^T w \]
$\lambda>0$ controls regularization strength.
\end{defn}

\begin{thm}[title=Ridge Regression Solution]
\[ w^\star = (\Phi^T\Phi + \lambda I_F)^{-1}\Phi^T y \]
\textbf{Multi-output:} $W^\star = (\Phi^T\Phi + \lambda I_F)^{-1}\Phi^T Y$
\end{thm}

% ============================================================
\section{Kernel Methods}
% ============================================================

\subsection{Kernel Function}
\begin{defn}[title=Kernel]
\[ k(x_i, x_j) = \phi(x_i)^T\phi(x_j) \]
Measures similarity without computing $\phi$ explicitly.
\end{defn}

\begin{itemize}
  \item \kv{Polynomial}{$(x_i^T x_j + c)^d$}
  \item \kv{Gaussian}{$\exp\!\left(-\frac{\|x_i-x_j\|^2}{2\sigma^2}\right)$}
\end{itemize}

\subsection{Kernel Matrix}
\[ K_{ij} = k(x_i, x_j) \in \R^{N\times N} \]

\subsection{Kernel Ridge Regression}
\begin{thm}[title=Kernel Ridge Solution]
\[ \alpha^\star = (K + \lambda I_N)^{-1} y \]
\end{thm}

\begin{thm}[title=Prediction]
\[ \hat{y} = y^T(K+\lambda I)^{-1}\,k(X,x) \]
where $k(X,x) = \bigl(k(x_1,x),\ldots,k(x_N,x)\bigr)^T$
\end{thm}

\textbf{Multi-output:}
\[ W^\star = \Phi^T(K+\lambda I)^{-1}Y, \quad \hat{y} = Y^T(K+\lambda I)^{-1}k(X,x) \]

\begin{warn}[title=Kernel selection]
Choose kernel by \textbf{cross-validation}. No general best choice.
\end{warn}
