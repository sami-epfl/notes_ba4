\section{Topic 1}

\subsection{Key definitions}
\begin{defn}[title=Linear map]
$T:V\to W$ is linear if $T(\alpha x+\beta y)=\alpha T(x)+\beta T(y)$
for all $x,y\in V$, $\alpha,\beta\in\R$.
\end{defn}

\begin{thm}[title=Rank--nullity]
For $T:V\to W$ with $\dim V<\infty$:
\[ \dim\ker T + \dim\operatorname{im}T \;=\; \dim V. \]
\end{thm}

\subsection{Formulas to remember}
\begin{itemize}
  \item \kv{Quadratic}{$x=\dfrac{-b\pm\sqrt{b^2-4ac}}{2a}$}
  \item \kv{Euler}{$e^{i\theta}=\cos\theta+i\sin\theta$}
  \item \kv{Geom.\ sum}{$\sum_{k=0}^{n}r^k=\dfrac{1-r^{n+1}}{1-r}$, $r\neq 1$}
  \item \kv{Taylor}{$f(x)=\sum_{k\ge0}\dfrac{f^{(k)}(a)}{k!}(x-a)^k$}
\end{itemize}

\begin{warn}[title=Pitfall]
$\log(ab)=\log a+\log b$ \emph{only} when $a,b>0$.
\end{warn}

\subsection{Useful identities}
\begin{align*}
 \sin^2\theta+\cos^2\theta &= 1 \\
 \cos(2\theta) &= \cos^2\theta-\sin^2\theta \\
 \sin(2\theta) &= 2\sin\theta\cos\theta
\end{align*}

\section{Topic 2 — Probability}

\subsection{Basics}
\begin{itemize}
  \item $\PR(A\cup B)=\PR(A)+\PR(B)-\PR(A\cap B)$
  \item $\PR(A\mid B)=\dfrac{\PR(A\cap B)}{\PR(B)}$
  \item Bayes: $\PR(A\mid B)=\dfrac{\PR(B\mid A)\PR(A)}{\PR(B)}$
\end{itemize}

\subsection{Common distributions}
\renewcommand{\arraystretch}{1.05}
\begin{tabularx}{\linewidth}{@{}lXl@{}}
\toprule
Dist. & PMF/PDF & $\E[X]$ \\
\midrule
Bern$(p)$  & $p^x(1-p)^{1-x}$           & $p$ \\
Bin$(n,p)$ & $\binom{n}{x}p^x(1-p)^{n-x}$ & $np$ \\
Poiss$(\lambda)$ & $e^{-\lambda}\lambda^x/x!$ & $\lambda$ \\
$\mathcal{N}(\mu,\sigma^2)$ & $\tfrac{1}{\sigma\sqrt{2\pi}}e^{-(x-\mu)^2/2\sigma^2}$ & $\mu$ \\
\bottomrule
\end{tabularx}

\begin{ex}[title=Worked]
$X\sim \mathcal{N}(0,1)$. Then $\PR(|X|<1)\approx 0.6827$.
\end{ex}

\subsection{Code snippet}
\begin{lstlisting}[language=Python]
def bsearch(a, x):
    lo, hi = 0, len(a)-1
    while lo <= hi:
        m = (lo+hi)//2
        if a[m] == x: return m
        if a[m] <  x: lo = m+1
        else:         hi = m-1
    return -1
\end{lstlisting}
